1   package dna.analysethis.service;
2   
3   import dna.analysethis.domain.Base;
4   import dna.analysethis.domain.Sequence;
5   import dna.analysethis.utilities.Manipulator;
6   import java.util.Collections;
7   
8   public class SequenceAnalysator {
9       private Sequence sequence;
10  
11      public SequenceAnalysator(Sequence sequence) {
12          this.sequence = sequence;
13      }
14      
15      public int frequency(Base b) {
16          return Collections.frequency(this.sequence.getSequence(), b);
17      }
18      
19      public int numberOfBases() {
20          return this.sequence.getSequence().size();
21      }
22      
23      public int numberOfUnknownBases() {
24          return this.frequency(Base.X);
25      }
26      
27      public double relativeFrequency(Base b) {
28          return (double) this.frequency(b) / this.numberOfBases();
29      }
30      
31      public double gcContent() {
32          return this.relativeFrequency(Base.G) + this.relativeFrequency(Base.C);
33      }
34      
35      public double mass(Base b) {
36          if (b == Base.X) {
37              throw new IllegalArgumentException("Tuntemattoman em√§ksen moolimassaa ei tunneta!");
38          }
39          
40          double mass = 0;
41          
42          if (b == Base.A) {
43              mass = 135.13;
44          } else if (b == Base.T) {
45              mass = 126.12;
46          } else if (b == Base.C) {
47              mass = 111.10;
48          } else if (b == Base.G) {
49              mass = 151.13;
50          }
51          
52          return this.frequency(b) * mass;
53      }
54      
55      public double totalMass() {
56          return this.mass(Base.C) + this.mass(Base.G) + this.mass(Base.T) + this.mass(Base.A);
57      }
58      
59      public double relativeMass(Base b) {
60          return this.mass(b) / this.totalMass();
61      }
62      
63      public double relativeGcMass() {
64          return this.relativeMass(Base.G) + this.relativeMass(Base.C);
65      }
66      
67      public int numberOfCodons() {
68          return Manipulator.sequenceToCodons(this.sequence).size();
69      }
70      
71      public boolean geneCheck() {
72          return this.numberOfCodons() >= 5 && this.relativeGcMass() >= .3; // && ...
73      }
74  }
