1   package dna.analysethis.service;
2   
3   import dna.analysethis.domain.Base;
4   import dna.analysethis.domain.Sequence;
5   import dna.analysethis.utilities.Manipulator;
6   import java.util.Collections;
7   import java.util.LinkedList;
8   import java.util.List;
9   
10  public class SequenceAnalysator {
11      private Sequence sequence;
12  
13      public SequenceAnalysator(Sequence sequence) {
14          this.sequence = sequence;
15      }
16      
17      public int frequency(Base b) {
18          return Collections.frequency(this.sequence.getSequence(), b);
19      }
20      
21      public int numberOfBases() {
22          return this.sequence.getSequence().size();
23      }
24      
25      public int numberOfUnknownBases() {
26          return this.frequency(Base.X);
27      }
28      
29      public double relativeFrequency(Base b) {
30          return (double) this.frequency(b) / this.numberOfBases();
31      }
32      
33      public double gcContent() {
34          return this.relativeFrequency(Base.G) + this.relativeFrequency(Base.C);
35      }
36      
37      public double mass(Base b) {
38          if (b == Base.X) {
39              throw new IllegalArgumentException("Tuntemattoman emäksen moolimassaa ei tunneta!");
40          }
41          
42          double mass = 0;
43          
44          if (b == Base.A) {
45              mass = 135.13;
46          } else if (b == Base.T) {
47              mass = 126.12;
48          } else if (b == Base.C) {
49              mass = 111.10;
50          } else if (b == Base.G) {
51              mass = 151.13;
52          }
53          
54          return this.frequency(b) * mass;
55      }
56      
57      public double mass() {
58          return this.mass(Base.C) + this.mass(Base.G) + this.mass(Base.T) + this.mass(Base.A);
59      }
60      
61      public double relativeMass(Base b) {
62          return this.mass(b) / this.mass();
63      }
64      
65      public double relativeGcMass() {
66          return this.relativeMass(Base.G) + this.relativeMass(Base.C);
67      }
68      
69      public int numberOfCodons() {
70          return Manipulator.sequenceToCodons(this.sequence).size();
71      }
72      
73      public boolean checkStartAndStopCodons(Sequence seq) {
74          Sequence startCodon = Manipulator.stringToSequence("ATG");
75          
76          List<Sequence> stopCodons = new LinkedList<>();
77          stopCodons.add(Manipulator.stringToSequence("TAA"));
78          stopCodons.add(Manipulator.stringToSequence("TAG"));
79          stopCodons.add(Manipulator.stringToSequence("TGA"));
80          
81          LinkedList<Sequence> codons = Manipulator.sequenceToCodons(seq);
82          
83          return codons.getFirst().equals(startCodon) && stopCodons.contains(codons.getLast());
84          // We omit checking for possible stop codons inside the sequence – for now that is
85      }
86      
87      public boolean checkStartAndStopCodons() {
88          // Multiple reading frames (4) make this especially difficult
89          return this.checkStartAndStopCodons(this.sequence)
90                  || this.checkStartAndStopCodons(Manipulator.reverse(this.sequence))
91                  || this.checkStartAndStopCodons(Manipulator.complement(this.sequence))
92                  || this.checkStartAndStopCodons(Manipulator.reverseComplement(this.sequence));
93      }
94      
95      public boolean checkIfGene() {
96          // Values used here are pulled out of a hat and have little scientific justification
97          return this.numberOfCodons() >= 5 && this.relativeGcMass() >= .3 && this.checkStartAndStopCodons();
98      }
99  }
